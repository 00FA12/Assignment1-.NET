90 private static IEnumerable<Claim> ParseClaimsFromJwt(string jwt)

92 string payload = jwt.Split('.')[1];

94 Dictionary<string, object>? keyValuePairs = JsonSerializer.Deserialize<Dictionary<string, object>>(jsonBytes);
95 return keyValuePairs!.Select(kvp => new Claim(kvp.Key, kvp.Value.ToString()!));
